topic_name_map_fuzzy = dict(zip(topic_info_df_fuzzy["topic"], topic_info_df_fuzzy["name"]))


def enrich_doc_topic_summary_with_names(doc_topic_summary_df: pd.DataFrame, topic_name_map: dict) -> pd.DataFrame:
    '''把多主题摘要中的 topic id 再翻译成可读主题名。'''
    rows = []
    for _, row in doc_topic_summary_df.iterrows():
        topic_ids = json.loads(row["topic_ids_json"])
        topic_weights = json.loads(row["topic_weights_json"])

        topic_names = [topic_name_map.get(int(topic_id), f"topic_{int(topic_id)}") for topic_id in topic_ids]
        topic_name_weight_pairs = [
            {
                "topic": int(topic_id),
                "name": topic_name_map.get(int(topic_id), f"topic_{int(topic_id)}"),
                "weight": float(weight),
            }
            for topic_id, weight in zip(topic_ids, topic_weights)
        ]

        payload = row.to_dict()
        payload["topic_names_json"] = json.dumps(topic_names, ensure_ascii=False)
        payload["topic_name_weight_pairs_json"] = json.dumps(topic_name_weight_pairs, ensure_ascii=False)
        rows.append(payload)

    return pd.DataFrame(rows)
